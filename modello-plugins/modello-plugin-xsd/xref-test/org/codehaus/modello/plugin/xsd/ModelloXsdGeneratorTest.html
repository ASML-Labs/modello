38  /**
39   * Check that features.mdo (which tries to be the most complete model) can be checked against XSD generated from
40   * Modello model <code>modello.mdo</code>.
41   *
42   * @author Herv√© Boutemy
43   */
44  public class ModelloXsdGeneratorTest
45      extends AbstractModelloGeneratorTest
46  {
47      public ModelloXsdGeneratorTest()
48      {
49          super( "modello" );
50      }
51  
52      public void testXsdGenerator()
53          throws Throwable
54      {
55          ModelloCore modello = (ModelloCore) lookup( ModelloCore.ROLE );
56  
57          Properties parameters = getModelloParameters( "1.4.0" );
58  
59          Model model = modello.loadModel( getTestFile( "../../src/main/mdo/modello.mdo" ) );
60  
61          modello.generate( model, "xsd", parameters );
62  
63          /* only available in JAXP 1.3, JDK 5+
64          SchemaFactory factory = SchemaFactory.newInstance( XMLConstants.W3C_XML_SCHEMA_NS_URI );
65          Schema schema = factory.newSchema( new StreamSource( new File( generatedSources, "features.xsd" ) ) );
66  
67          SAXParserFactory spf = SAXParserFactory.newInstance();
68          spf.setSchema( schema );
69          SAXParser parser = spf.newSAXParser();
70          parser.parse( new InputSource( getClass().getResourceAsStream( "/features.xml" ) ) );
71          */
72  
73          SAXParserFactory factory = SAXParserFactory.newInstance();
74          factory.setValidating( true );
75          factory.setNamespaceAware( true );
76          SAXParser saxParser = factory.newSAXParser();
77          saxParser.setProperty( "http://java.sun.com/xml/jaxp/properties/schemaLanguage",
78                                 "http://www.w3.org/2001/XMLSchema" );
79          saxParser.setProperty( "http://java.sun.com/xml/jaxp/properties/schemaSource",
80                                 new File( getOutputDirectory(), "modello-1.4.0.xsd" ) );
81  
82          // first self-test: validate Modello model with xsd generated from it
83          saxParser.parse( getTestFile( "../../src/main/mdo/modello.mdo" ), new Handler() );
84  
85          // then features.mdo
86          saxParser.parse( getClass().getResourceAsStream( "/features.mdo" ), new Handler() );
87      }
88  
89      private static class Handler
90          extends DefaultHandler
91      {
92          public void warning ( SAXParseException e )
93              throws SAXException
94          {
95              throw e;
96          }
97  
98          public void error ( SAXParseException e )
99              throws SAXException
100         {
101             throw e;
102         }
103     }
